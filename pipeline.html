process = function(dat){
      # saving gene names
      mapping = as.numeric(dat[, 1])
      # removing gene names from main matrix
      hvals = (dat[, -1])
      # gene names now row names
      rownames(hvals) = mapping
      hvals = as.matrix(hvals)
      hvals = scale(hvals)
      return(hvals)
}
